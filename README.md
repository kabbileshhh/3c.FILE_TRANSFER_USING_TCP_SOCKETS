# 3c.CREATION FOR FILE TRANSFER USING TCP SOCKETS
## AIM
To write a python program for creating File Transfer using TCP Sockets Links
## ALGORITHM:
1. Import the necessary python modules.
2. Create a socket connection using socket module.
3. Send the message to write into the file to the client file.
4. Open the file and then send it to the client in byte format.
5. In the client side receive the file from server and then write the content into it.
## PROGRAM
### Server:
```
import os
import socket

port = 60001
s = socket.socket()

host = socket.gethostname()

s.bind((host, port))


s.listen(5)

while True:
    conn, addr = s.accept()
    
    data = conn.recv(1024)
    
    print('Server received', repr(data))
    
    filename = 'mytext.txt'
    
    if not os.path.exists(filename):
        with open(filename, 'w', encoding='utf-8') as sample_file:
            sample_file.write('This is a sample file generated by the server.\n')
            sample_file.write('Replace mytext.txt with your own file if needed.\n')
        print(f'Created missing file: {filename}')
    
    with open(filename, 'rb') as f:
        l = f.read(1024)
        
        while l:
            conn.send(l)
            
            print('Sent ', repr(l))
            
            l = f.read(1024)
    
    print('Done sending')
    
    conn.send('Thank you for connecting'.encode())
    
    conn.close()
```
### Client:
```
import socket
s = socket.socket()
host = socket.gethostname()
port = 60001
s.connect((host, port))
s.send("Hello server!".encode())
with open('received_file', 'wb') as f:
    while True:
        print('receiving data...')
        data = s.recv(1024)
        print('data=%s', (data))
        if not data:
            break
        f.write(data)
f.close()
print('Successfully get the file')
s.close()
print('connection closed')
```
